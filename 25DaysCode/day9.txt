class Solution{
    public:
    // Function to find equilibrium point in the array.
    // a: input array
    // n: size of array
    int equilibriumPoint(long long arr[], int n) {
    
       
       int sum=0;
       
        for(int i=0; i<n; i++)
            sum += arr[i];
            
            
            
        int leftsum=0;
        int rightsum=0;
        
        
        for(int j=0; j<n; j++)
        {
            rightsum = sum-arr[j]-leftsum;
            
            
            if(rightsum == leftsum)
                return j+1;
                
            leftsum += arr[j];
        }
        
        return -1;
    }

};
